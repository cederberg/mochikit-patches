 22 /**
 23  * Returns a right-truncated copy of a string or an array. If the
 24  * string or array is shorter than the specified maximum length, the
 25  * object will be returned unmodified. If an optional tail string or
 26  * array is specified, additional elements will be removed in order
 27  * to also make space for the tail (that will be appended).
 28  *
 29  * @param {String/Array} obj the string or array to truncate
 30  * @param {Number} maxLength the maximum length
 31  * @param {String/Array} [tail] the tail to append on truncation
 32  *
 33  * @return {String/Array} the truncated string or array
 34  */
 35 MochiKit.Format.truncate = function(obj, maxLength, tail) {
 36     var base = MochiKit.Base;
 37     if (obj != null && typeof(obj) != "string" && !base.isArrayLike(obj)) {
 38         obj = obj.toString();
 39     }
 40     if (obj == null || obj.length <= maxLength || maxLength < 0) {
 41         return obj;
 42     }
 43     if (typeof(tail) == "string" || base.isArrayLike(tail)) {
 44         obj = obj.slice(0, maxLength - tail.length);
 45         if (typeof(obj) == "string") {
 46             return obj + tail;
 47         } else {
 48             return base.extend(obj, tail);
 49         }
 50     } else {
 51         return obj.slice(0, maxLength);
 52     }
 53 }

 55 /**
 56  * Creates a formatter function for the specified formatter pattern
 57  * and locale. The returned function takes as many arguments as the
 58  * formatter pattern requires. See separate documentation for
 59  * information about the formatter pattern syntax.
 60  *
 61  * @param {String} pattern the formatter pattern string
 62  * @param {Object} [locale] the locale to use, defaults to
 63  *            LOCALE.en_US
 64  *
 65  * @return {Function} the formatter function created
 66  *
 67  * @throws FormatPatternError if the format pattern was invalid
 68  */
 69 MochiKit.Format.formatter = function(pattern, locale) {
 70     if (typeof(locale) == "undefined") {
 71         locale = MochiKit.Format.formatLocale();
 72     }
 73     var parts = MochiKit.Format._parsePattern(pattern);
 74     return function() {
 75         var values = MochiKit.Base.extend([], arguments);
 76         return MochiKit.Format._formatParts(parts, values, locale);
 77     }
 78 }

 80 /**
 81  * Formats the specified arguments according to a formatter pattern.
 82  * See separate documentation for information about the formatter
 83  * pattern syntax.
 84  *
 85  * @param {String} pattern the formatter pattern string
 86  * @param {Object} [...] the optional values to format
 87  *
 88  * @return {String} the formatted output string
 89  *
 90  * @throws FormatPatternError if the format pattern was invalid
 91  */
 92 MochiKit.Format.format = function(pattern/*, ...*/) {
 93     var parts = MochiKit.Format._parsePattern(pattern);
 94     var values = MochiKit.Base.extend([], arguments, 1);
 95     var locale = MochiKit.Format.formatLocale();
 96     return MochiKit.Format._formatParts(parts, values, locale);
 97 }

 99 /**
100  * Parses a format pattern and returns an array of constant strings
101  * and format info objects.
102  *
103  * @param {String} pattern the format pattern to analyze
104  *
105  * @return {Array} an array of strings and format info objects
106  *
107  * @throws FormatPatternError if the format pattern was invalid
108  */
109 MochiKit.Format._parsePattern = function(pattern) {
110     var self = MochiKit.Format;
111     var parts = [];
112     var start = 0;
113     var pos = 0;
114     for (pos = 0; pos < pattern.length; pos++) {
115         if (pattern[pos] == "{") {
116             if (pos + 1 >= pattern.length) {
117                 var msg = "unescaped { char, should be escaped as {{";
118                 throw new self.FormatPatternError(pattern, pos, msg);
119             } else if (pattern[pos + 1] == "{") {
120                 parts.push(pattern.substring(start, pos + 1));
121                 start = pos + 2;
122                 pos++;
123             } else {
124                 if (start < pos) {
125                     parts.push(pattern.substring(start, pos));
126                 }
127                 start = pattern.indexOf("}", pos) + 1;
128                 if (start <= 0) {
129                     var msg = "unmatched { char, not followed by a } char";
130                     throw new self.FormatPatternError(pattern, pos, msg);
131                 }
132                 parts.push(self._parseFormat(pattern, pos + 1, start - 1));
133                 pos = start - 1;
134             }
135         } else if (pattern[pos] == "}") {
136             if (pos + 1 >= pattern.length || pattern[pos + 1] != "}") {
137                 var msg = "unescaped } char, should be escaped as }}";
138                 throw new self.FormatPatternError(pattern, pos, msg);
139             }
140             parts.push(pattern.substring(start, pos + 1));
141             start = pos + 2;
142             pos++;
143         }
144     }
145     if (start < pos) {
146         parts.push(pattern.substring(start, pos));
147     }
148     return parts;
149 }

151 /**
152  * Parses a format instruction and returns a format info object.
153  *
154  * @param {String} pattern the format pattern string
155  * @param {Number} startPos the first index of the format instruction
156  * @param {Number} endPos the last index of the format instruction
157  *
158  * @return {Object} the format info object
159  *
160  * @throws FormatPatternError if the format pattern was invalid
161  */
162 MochiKit.Format._parseFormat = function(pattern, startPos, endPos) {
163     var self = MochiKit.Format;
164     var text = pattern.substring(startPos, endPos);
165     var info;
166     var pos = text.indexOf(":");
167     if (pos == 0) {
168         info = self._parseFormatFlags(pattern, startPos + 1, endPos);
169         info.path = [0];
170     } else if (pos > 0) {
171         info = self._parseFormatFlags(pattern, startPos + pos + 1, endPos);
172         info.path = text.substring(0, pos).split(".");
173     } else {
174         info = self._parseFormatFlags(pattern, endPos, endPos);
175         info.path = text.split(".");
176     }
177     var DIGITS = /^\d+$/;
178     for (var i = 0; i < info.path.length; i++) {
179         var e = info.path[i];
180         if (typeof(e) == "string") {
181             e = self.strip(e);
182             if (e == "" && info.path.length == 1) {
183                 e = 0;
184             } else if (e == "") {
185                 var msg = "format value path contains blanks";
186                 throw new self.FormatPatternError(pattern, startPos, msg);
187             } else if (DIGITS.test(e)) {
188                 e = parseInt(e);
189             }
190         }
191         info.path[i] = e;
192     }
193     if (info.path.length < 0 || typeof(info.path[0]) != "number") {
194         info.path.unshift(0);
195     }
196     return info;
197 }

199 /**
200  * Parses a string with format flags and returns a format info object.
201  *
202  * @param {String} pattern the format pattern string
203  * @param {Number} startPos the first index of the format instruction
204  * @param {Number} endPos the last index of the format instruction
205  *
206  * @return {Object} the format info object
207  *
208  * @throws FormatPatternError if the format pattern was invalid
209  */
210 MochiKit.Format._parseFormatFlags = function(pattern, startPos, endPos) {
211     var self = MochiKit.Format;
212     var info = { format: "s", width: 0, precision: -1, align: ">",
213                  sign: "-", padding: " ", grouping: false };
214     var flags = self.rstrip(pattern.substring(startPos, endPos));
215     while (flags.length > 0) {
216         switch (flags[0]) {
217         case ">":
218         case "<":
219             info.align = flags[0];
220             flags = flags.substring(1);
221             break;
222         case "+":
223         case "-":
224         case " ":
225             info.sign = flags[0];
226             flags = flags.substring(1);
227             break;
228         case ",":
229             info.grouping = true;
230             flags = flags.substring(1);
231             break;
232         case ".":
233             var DIGITS = "0123456789";
234             var pos = 1;
235             while (pos < flags.length && DIGITS.indexOf(flags[pos]) >= 0) {
236                 pos++;
237             }
238             info.precision = parseInt(flags.substring(1, pos));
239             flags = flags.substring(pos);
240             break;
241         case "0":
242             info.padding = flags[0];
243             flags = flags.substring(1);
244             break;
245         case "1":
246         case "2":
247         case "3":
248         case "4":
249         case "5":
250         case "6":
251         case "7":
252         case "8":
253         case "9":
254             var DIGITS = "0123456789";
255             var pos = 1;
256             while (pos < flags.length && DIGITS.indexOf(flags[pos]) >= 0) {
257                 pos++;
258             }
259             info.width = parseInt(flags.substring(0, pos));
260             flags = flags.substring(pos);
261             break;
262         case "s":
263         case "r":
264         case "b":
265         case "c":
266         case "d":
267         case "o":
268         case "x":
269         case "X":
270         case "f":
271         case "%":
272             info.format = flags[0];
273             flags = flags.substring(1);
274             break;
275         default:
276             var msg = "unsupported format flag: " + flags[0];
277             throw new self.FormatPatternError(pattern, startPos, msg);
278         }
279     }
280     return info;
281 }

283 /**
284  * Formats an array of strings and format info objects with the
285  * specified array of values.
286  *
287  * @param {Array} parts the array of strings and format infos
288  * @param {Array} values the array of values to format
289  * @param {Object} locale the formatting locale to use
290  *
291  * @return {String} the formatted output string
292  */
293 MochiKit.Format._formatParts = function(parts, values, locale) {
294     var self = MochiKit.Format;
295     var result = "";
296     for (var i = 0; i < parts.length; i++) {
297         if (typeof(parts[i]) == "string") {
298             result += parts[i];
299         } else {
300             var info = parts[i];
301             var v = values;
302             for (var j = 0; j < info.path.length; j++) {
303                 if (v != null) {
304                     v = v[info.path[j]];
305                 }
306             }
307             var str = "";
308             switch (info.format) {
309             // TODO: implement remaining format types (with precision)
310             case "d":
311             case "f":
312                 var sign = (info.sign == "-") ? "" : info.sign;
313                 sign = (v < 0) ? "-" : sign;
314                 v = Math.abs(v);
315                 if (info.format == "d") {
316                     str = self.truncToFixed(v, 0);
317                 } else if (info.precision >= 0) {
318                     str = self.truncToFixed(v, info.precision);
319                 } else {
320                     str = (v == null) ? "0" : v.toString();
321                 }
322                 var fracPos = str.indexOf(".");
323                 var whole = (fracPos < 0) ? str : str.substring(0, fracPos);
324                 var fraction = (fracPos < 0) ? "" : str.substring(fracPos + 1);
325                 if (info.padding == "0" ) {
326                     var adjust = sign.length;
327                     if (fraction.length > 0) {
328                         adjust += 1 + fraction.length;
329                     }
330                     while (whole.length < info.width - adjust) {
331                         whole = "0" + whole;
332                     }
333                 }
334                 str = (fraction.length > 0) ? locale.decimal : "";
335                 while (info.grouping && fraction.length > 3) {
336                     str = str + fraction.substring(0, 3) + locale.separator;
337                     fraction = fraction.substring(3);
338                     if (whole[0] == "0") {
339                         whole = whole.substring(1);
340                     }
341                 }
342                 if (fraction.length > 0) {
343                     str += fraction;
344                 }
345                 while (info.grouping && whole.length > 3) {
346                     var pos = whole.length - 3;
347                     str = locale.separator + whole.substring(pos) + str;
348                     if (whole[0] == "0") {
349                         whole = whole.substring(1, pos);
350                     } else {
351                         whole = whole.substring(0, pos);
352                     }
353                 }
354                 str = sign + whole + str;
355                 break;
356             case "r":
357             case "s":
358             default:
359                 if (info.format == "r") {
360                     str = MochiKit.Base.repr(v);
361                 } else {
362                     str = (v == null) ? "null" : v.toString();
363                 }
364                 str = self.truncate(str, info.precision);
365                 break;
366             }
367             while (info.width > str.length) {
368                 if (info.align == "<") {
369                     str += info.padding;
370                 } else {
371                     str = info.padding + str;
372                 }
373             }
374             result += str;
375         }
376     }
377     return result;
378 }

380 /**
381  * Creates a new format pattern error.
382  *
383  * @param {String} pattern the format pattern string
384  * @param {Number} pos the position of the error
385  * @param {String} message the error message text
386  *
387  * @return {Error} the format pattern error
388  *
389  * @class The format pattern error class. This error is thrown when
390  *     a syntax error is encountered inside a format string.
391  * @property {String} pattern The format pattern string.
392  * @property {Number} pos The position of the error.
393  * @property {String} message The error message text.
394  * @extends MochiKit.Base.NamedError
395  */
396 MochiKit.Format.FormatPatternError = function(pattern, pos, message) {
397     this.pattern = pattern;
398     this.pos = pos;
399     this.message = message;
400 }
401 MochiKit.Format.FormatPatternError.prototype =
402     new MochiKit.Base.NamedError("MochiKit.Format.FormatPatternError");
