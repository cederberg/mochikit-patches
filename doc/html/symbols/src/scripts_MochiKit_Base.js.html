 22 /**
 23  * Checks if the specified value corresponds to false. This function
 24  * will equate false, undefined, null, 0, "", [], "false" and "null"
 25  * with a boolean false value.
 26  *
 27  * @param {Object} value the value to check
 28  *
 29  * @return {Boolean} true if the value corresponds to false, or
 30  *         false otherwise
 31  */
 32 MochiKit.Base.isFalse = function(value) {
 33     return value == false || value == null || value == 0 ||
 34            value.length === 0 || value == "false" || value == "null";
 35 }

 37 /**
 38  * Returns the first function argument that is not undefined.
 39  *
 40  * @param {Object} [...] the values to check
 41  *
 42  * @return {Object} the first non-undefined argument, or
 43  *         undefined if all arguments were undefined 
 44  */
 45 MochiKit.Base.defaultValue = function(/* ... */) {
 46     for (var i = 0; i < arguments.length; i++) {
 47         if (typeof(arguments[i]) != "undefined") {
 48             return arguments[i];
 49         }
 50     }
 51     return undefined;
 52 }

 54 /**
 55  * Creates a dictionary object from a list of keys and values. It
 56  * can be used either as a reverse of items(), or as a reverse of
 57  * keys() and values(). That is, either the function take a single
 58  * list where each element contains both key and value, or it takes
 59  * two separate lists, one with keys and the other with values. If
 60  * a key is specified twice, only the last value will be used.
 61  *
 62  * @code dict([['a', 1], ['b', 2]]) --> { a: 1, b: 2 }
 63  *       dict(['a','b'], [1, 2]) --> { a: 1, b: 2 }
 64  *
 65  * @param {Array} itemsOrKeys the list of items or keys
 66  * @param {Array} [values] the optional list of values
 67  *
 68  * @return {Object} a dictionary object with all the keys set to the
 69  *         corresponding value
 70  */
 71 MochiKit.Base.dict = function(itemsOrKeys, values) {
 72     var o = {};
 73     if (!MochiKit.Base.isArrayLike(itemsOrKeys)) {
 74         throw new TypeError("First argument must be array-like");
 75     }
 76     if (MochiKit.Base.isArrayLike(values) && itemsOrKeys.length !== values.length) {
 77         throw new TypeError("Both arrays must be of same length");
 78     }
 79     for (var i = 0; i < itemsOrKeys.length; i++) {
 80         var k = itemsOrKeys[i];
 81         if (k === null || k === undefined) {
 82             throw new TypeError("Key at index " + i + " is null or undefined");
 83         } else if (MochiKit.Base.isArrayLike(k)) {
 84             o[k[0]] = k[1];
 85         } else if (MochiKit.Base.isArrayLike(values)) {
 86             o[k] = values[i];
 87         } else {
 88             o[k] = values;
 89         }
 90     }
 91     return o;
 92 }

 94 /**
 95  * Creates a new object by copying keys and values from another
 96  * object. A list of key names (or an object whose property names
 97  * will be used as keys) must be specified as an argument. The
 98  * returned object will only contain properties that were defined in
 99  * the source object, keeping the source object values. The source
100  * object will be left unmodified.
101  *
102  * @param {Object} src the source object to select values from
103  * @param {Array/Object} keys the list of keys to select, or an
104  *            object with the keys to select
105  *
106  * @return {Object} a new object containing the matching keys and
107  *             values found in the source object
108  */
109 MochiKit.Base.select = function(src, keys) {
110     var res = {};
111     if (!MochiKit.Base.isArrayLike(keys)) {
112         keys = MochiKit.Base.keys(keys);
113     }
114     for (var i = 0; i < keys.length; i++) {
115         var k = keys[i];
116         if (k in src) {
117             res[k] = src[k];
118         }
119     }
120     return res;
121 }

123 /**
124  * Filters an object by removing a list of keys. A list of key names
125  * (or an object whose property names will be used as keys) must be
126  * specified as an argument. A new object containing the source
127  * object values for the specified keys will be returned. The source
128  * object will be modified by removing all the specified keys.
129  *
130  * @param {Object} src the source object to select and modify
131  * @param {Array/Object} keys the list of keys to remove, or an
132  *            object with the keys to remove
133  *
134  * @return {Object} a new object containing the matching keys and
135  *             values found in the source object
136  */
137 MochiKit.Base.mask = function(src, keys) {
138     var res = {};
139     if (!MochiKit.Base.isArrayLike(keys)) {
140         keys = MochiKit.Base.keys(keys);
141     }
142     for (var i = 0; i < keys.length; i++) {
143         var k = keys[i];
144         if (k in src) {
145             res[k] = src[k];
146             delete src[k];
147         }
148     }
149     return res;
150 }

152 /**
153  * Returns a truncated copy of a string or an array. If the string
154  * or array is shorter than the specified maximum length, the object
155  * will be returned unmodified. If an optional tail string or array
156  * is specified, additional elements will be removed from the object
157  * to append it to the end.
158  *
159  * @param {String/Array} obj the string or array to truncate
160  * @param {Number} maxLength the maximum length
161  * @param {String/Array} tail the optional tail to use on truncation
162  *
163  * @return {String/Array} the truncated string or array
164  */
165 MochiKit.Base.truncate = function(obj, maxLength, tail) {
166     if (obj != null && typeof(obj) != "string" && !MochiKit.Base.isArrayLike(obj)) {
167         obj = obj.toString();
168     }
169     if (obj == null || obj.length <= maxLength) {
170         return obj;
171     }
172     if (typeof(tail) == "string" || MochiKit.Base.isArrayLike(tail)) {
173         obj = obj.slice(0, maxLength - tail.length);
174         if (typeof(obj) == "string") {
175             return obj + tail;
176         } else {
177             return MochiKit.Base.extend(obj, tail);
178         }
179     } else {
180         return obj.slice(0, maxLength);
181     }
182 }

184 /**
185  * Returns the name of a function. This is often useful for debugging
186  * or logging purposes. If the function is anonymous or the
187  * JavaScript environment doesn't provide function <code>name</code>
188  * properties, any registered function name or undefined will be
189  * returned.
190  *
191  * @param {Function} func the function to name
192  *
193  * @return {String} the function name, or undefined if not available
194  *
195  * @see MochiKit.Base.registerFunctionNames
196  */
197 MochiKit.Base.functionName = function(func) {
198     if (func == null) {
199         return null;
200     } else if (func.name != null && func.name != "") {
201         return func.name;
202     } else {
203         return func.NAME;
204     }
205 }

207 /**
208  * Registers function names for debugging or logging. This is useful
209  * when using anonymous functions or inside JavaScript environments
210  * that do not provide function <code>name</code> properties. This
211  * function will add the specified name as a new <code>NAME</code>
212  * property to any function that doesn't already have a name. This
213  * function will also process any properties or prototype properties
214  * recursively adding names like <code>name.[property name]</code>.
215  *
216  * @param {Object} obj the function or object to register
217  * @param {String} name the function or object (class) name
218  * @param {Array} [stack] the object stack to avoid circular recursion
219  *
220  * @see MochiKit.Base.functionName
221  */
222 MochiKit.Base.registerFunctionNames = function(obj, name, stack) {
223    if (typeof(obj) === "function" &&
224        (obj.name == null || obj.name == "") &&
225        typeof(obj.NAME) === "undefined") {
226        obj.NAME = name;
227    }
228    stack = stack || [];
229    if (obj != null && name != null &&
230        (typeof(obj) === "object" || typeof(obj) === "function") &&
231        obj !== Object.prototype && obj !== Function.prototype &&
232        typeof(obj.nodeType) !== "number" &&
233        MochiKit.Base.findIdentical(stack, obj) < 0) {
234 
235        stack.push(obj);
236        for (var prop in obj) {
237            var str = name + "." + prop;
238            MochiKit.Base.registerFunctionNames(obj[prop], str, stack);
239        }
240        var str = name + ".prototype";
241        MochiKit.Base.registerFunctionNames(obj.prototype, str, stack);
242        stack.pop();
243    }
244 }

246 /**
247  * Returns the current execution stack trace. The stack trace is an
248  * array of function names with the innermost function at the lowest
249  * index (0). Due to limitations in the JavaScript API:s, the stack
250  * trace will be cut if recursion is detected. The stack trace will
251  * also be cut if the call depth exceeds the maximum depth or if any
252  * function in the chain has an injected stack trace.
253  *
254  * @param {Number} [maxDepth] the maximum call depth, defaults to 20
255  *
256  * @return {Array} the stack trace array of function names
257  *
258  * @see MochiKit.Base.functionName
259  * @see MochiKit.Base.injectStackTrace
260  */
261 MochiKit.Base.stackTrace = function(maxDepth) {
262     var func = arguments.callee.caller;
263     var visited = [];
264     var res = [];
265     maxDepth = maxDepth || 20;
266     while (func != null) {
267         if (MochiKit.Base.findIdentical(visited, func) >= 0) {
268             res.push("...recursion...");
269             break;
270         }
271         if (func.$stackTrace != null) {
272             res = res.concat(func.$stackTrace);
273             break;
274         }
275         var name = MochiKit.Base.functionName(func);
276         if (name === null) {
277             // Skip stack trace when null (but not when undefined)
278         } else {
279             res.push(name || "<anonymous>");
280         }
281         visited.push(func);
282         if (visited.length >= maxDepth) {
283             res.push("...");
284             break;
285         }
286         func = func.caller;
287     }
288     return res;
289 }

291 /**
292  * Injects a stack trace for a function. This method is useful for
293  * creating a fake stack trace in anonymous or callback functions. A
294  * null value can be used to clear any previously injected stack
295  * trace for the calling function.
296  *
297  * @param {Array} stackTrace the stack trace, or null to clear
298  * @param {Function} [func] the function to modify, or null for the
299  *            currently executing function (i.e. the caller)
300  */
301 MochiKit.Base.injectStackTrace = function(stackTrace, func) {
302     func = func || arguments.callee.caller;
303     if (func != null) {
304         if (stackTrace) {
305             func.$stackTrace = stackTrace;
306         } else {
307             delete func.$stackTrace;
308         }
309     }
310 }
